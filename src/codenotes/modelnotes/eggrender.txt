        if (tileClass != null)
        {
            TileEntitySpecialRenderer<?> r = TileEntityRendererDispatcher.instance.getRenderer(tileClass);
            if (r != null)
            {
                r.render(null, 0, 0, 0, 0, -1, 0.0F);

        }
